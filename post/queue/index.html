队列实现

March 17, 2023 in golang, server queue

本文主要实现自定义的几种队列，代码实现如下：

使用golang的list

使用list自带的PushBack(),PushFront()等特性实现enqueue

package main

import (
	"container/list"
	"fmt"
	"testing"
)

func TestList(_ *testing.T) {
	l := list.New()
	l.PushFront("10")
	l.PushFront("20")
	l.PushBack("40")
	l.PushBack("50") // 20 10 40 50

	fmt.Println("len -- ", l.Len())
	front := l.Front() // 20
	fmt.Println(front.Value)

	for e := l.Front(); e != nil; e = e.Next() {
		fmt.Printf("%T %v \t", e.Value, e.Value) // string 20 	string 10 	string 40 	string 50
	}
}

func TestList2(_ *testing.T) {
	customQueue := &customQueue{
		queue: list.New(),
	}
	fmt.Printf("Enqueue: A\n")
	customQueue.Enqueue("A")
	fmt.Printf("Enqueue: B\n")
	customQueue.Enqueue("B")
	fmt.Printf("Size: %d\n", customQueue.Size())
	for customQueue.Size() > 0 {
		frontVal, _ := customQueue.Front()
		fmt.Printf("Front: %s\n", frontVal)
		fmt.Printf("Dequeue: %s\n", frontVal)
		customQueue.Dequeue()
	}
	fmt.Printf("Size: %d\n", customQueue.Size())
}

type customQueue struct {
	queue *list.List
}

func (c *customQueue) Enqueue(value string) {
	c.queue.PushBack(value)
}

func (c *customQueue) Dequeue() error {
	if c.queue.Len() > 0 {
		ele := c.queue.Front()
		c.queue.Remove(ele)
	}
	return fmt.Errorf("pop Error: Queue is empty")
}

func (c *customQueue) Front() (string, error) {
	if c.queue.Len() > 0 {
		if val, ok := c.queue.Front().Value.(string); ok {
			return val, nil
		}
		return "", fmt.Errorf("peep Error: Queue Datatype is incorrect")
	}
	return "", fmt.Errorf("peep Error: Queue is empty")
}

func (c *customQueue) Size() int {
	return c.queue.Len()
}

func (c *customQueue) Empty() bool {
	return c.queue.Len() == 0
}

package main

import (
	"fmt"
	"math"
	"sync"
	"testing"
	"time"
)

func TestCh(t *testing.T) {
	queue := NewQueue(3)
	for i := 0; i < 10; i++ {
		err := queue.Enqueue(i)
		if err != nil {
			t.Fatal(err)
			return
		}
	}

	for i := 0; i < 3; i++ {
		t.Log(queue.Dequeue())
	}
	time.Sleep(1 * time.Second)
	t.Log("cache ---- ", queue.CacheSize())
}

type MyChannelQueue struct {
	capacity   int32
	list       chan interface{}
	tmpStorage []interface{}
	mutex      sync.Mutex
}

func NewQueue(size int32) *MyChannelQueue {
	return &MyChannelQueue{
		capacity:   size,
		list:       make(chan interface{}, size),
		tmpStorage: make([]interface{}, 0, size),
	}
}

const maxQueueSize = math.MaxInt32

func (q *MyChannelQueue) Enqueue(item interface{}) error {
	if len(q.tmpStorage) == maxQueueSize { // max cache is maxQueueSize
		return fmt.Errorf("more than max queue size")
	}
	select {
	case q.list <- item:
	default: // if the list already full, save the element to the cache
		q.mutex.Lock()
		q.tmpStorage = append(q.tmpStorage, item)
		q.mutex.Unlock()
	}
	return nil
}

func (q *MyChannelQueue) Dequeue() interface{} {
	item := <-q.list
	go func() {
		q.flushCache()
	}()
	return item
}

func (q *MyChannelQueue) flushCache() {
	for {
		q.mutex.Lock()
		if len(q.tmpStorage) == 0 {
			q.mutex.Unlock()
			break
		}
		select {
		case q.list <- q.tmpStorage[0]: // when list is spare
			q.tmpStorage = q.tmpStorage[1:] // update cache into list
		default:
			q.mutex.Unlock()
			continue
		}
		q.mutex.Unlock()
	}
}

func (q *MyChannelQueue) Size() int {
	q.mutex.Lock()
	defer q.mutex.Unlock()
	return len(q.list)
}

func (q *MyChannelQueue) CacheSize() int {
	q.mutex.Lock()
	defer q.mutex.Unlock()
	return len(q.tmpStorage)
}

使用切片

package main

import (
	"fmt"
	"testing"
)

type StringQueue struct {
	queue []string
	size  int64
}

func (q *StringQueue) Enqueue(s string) error {
	if q.size == 0 {
		return fmt.Errorf("invalid capacity")
	}
	if int64(len(q.queue)) == q.size {
		return fmt.Errorf("no space")
	}

	q.queue = append(q.queue, s)
	return nil
}

func (q *StringQueue) Dequeue() string {
	if len(q.queue) == 0 {
		return ""
	}
	temp := q.queue[0]
	q.queue = q.queue[1:]
	return temp
}

func (q *StringQueue) Empty() bool {
	return len(q.queue) == 0
}

func (q *StringQueue) Top() string {
	return q.queue[0]
}

func TestSliceQueue(t *testing.T) {
	queue := StringQueue{make([]string, 0), 20}
	(&queue).Enqueue("Item 1") //nolint:errcheck
	(&queue).Enqueue("Item 2") //nolint:errcheck
	t.Log("Top:", (&queue).Top())
	t.Log("Dequeued", (&queue).Dequeue())
	t.Log("New Top:", (&queue).Top())
	t.Log("Empty?", (&queue).Empty())
	t.Log("Dequeued", (&queue).Dequeue())
	t.Log("Empty now?", (&queue).Empty())
}
